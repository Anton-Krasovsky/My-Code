fun main() {
    val greeting = "I love animals!\n" +
            "Let's check on the animals..."
    println(greeting)
    while (true) {
        println("Please enter the number of the habitat you would like to view: ")
        when (readln()) {
            "1" -> println(animalZoo.camel.nameAnimal)
            "2" -> println(animalZoo.lion.nameAnimal)
            "3" -> println(animalZoo.deer.nameAnimal)
            "4" -> println(animalZoo.goose.nameAnimal)
            "5" -> println(animalZoo.bat.nameAnimal)
            "6" -> println(animalZoo.rabbit.nameAnimal)
            "exit" -> break
        }
        println("You've reached the end of the program. To check another habitat, please restart the watcher.")
    }
    println("See you later")
    }
